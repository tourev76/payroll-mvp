    <script>

        let employees = [];

        let attendanceRecords = [];

        function switchTab(tabName) {

            document.querySelectorAll('.section').forEach(div => div.style.display = 'none');

            document.getElementById(tabName).style.display = 'block';

            document.querySelectorAll('.tab').forEach(t => t.classList.remove('active-tab'));

            event.target.classList.add('active-tab');

        }

        function addEmployee() {

            const employee = {

                id: Date.now(),

                name: document.getElementById('name').value,

                position: document.getElementById('position').value,

                contractType: document.getElementById('contractType').value,

                salary: parseFloat(document.getElementById('salary').value)

            };

           

            employees.push(employee);

            updateEmployeeLists();

            document.getElementById('employeeForm').reset();

        }

        function updateEmployeeLists() {

            // Update employee table

            const table = document.getElementById('employeeList');

            table.innerHTML = `<tr>

                <th>Nom</th>

                <th>Poste</th>

                <th>Type de Contrat</th>

                <th>Salaire</th>

            </tr>`;

           

            employees.forEach(emp => {

                table.innerHTML += `<tr>

                    <td>${emp.name}</td>

                    <td>${emp.position}</td>

                    <td>${emp.contractType}</td>

                    <td>${emp.salary.toFixed(2)} XOF</td>

                </tr>`;

            });

            // Update dropdowns

            const options = employees.map(emp =>

                `<option value="${emp.id}">${emp.name} (${emp.position})</option>`

            ).join('');

            document.getElementById('employeeSelect').innerHTML = options;

            document.getElementById('payrollEmployee').innerHTML = options;

        }

        function calculatePayroll() {

            const employeeId = parseInt(document.getElementById('payrollEmployee').value);

            const employee = employees.find(emp => emp.id === employeeId);

            const attendance = attendanceRecords.find(rec => rec.id === employeeId) || {hours: 0, overtime: 0};

            // Simple calculations (replace with actual Senegal-specific formulas)

            const overtimePay = attendance.overtime * (employee.salary / 160) * 1.5;

            const gross = employee.salary + overtimePay;

            const tax = gross * 0.1; // Simplified tax

            const css = gross * 0.07; // Employee CSS contribution

            const net = gross - tax - css;

            // Display results

            document.getElementById('psName').textContent = employee.name;

            document.getElementById('baseSalary').textContent = employee.salary.toFixed(2);

            document.getElementById('overtimePay').textContent = overtimePay.toFixed(2);

            document.getElementById('grossPay').textContent = gross.toFixed(2);

            document.getElementById('tax').textContent = tax.toFixed(2);

            document.getElementById('css').textContent = css.toFixed(2);

            document.getElementById('netPay').textContent = net.toFixed(2);

        }
